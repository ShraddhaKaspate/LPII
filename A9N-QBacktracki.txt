# N-Queens Problem using Backtracking

N = 4
board = [-1] * N

# Check safe position
def check(row, col):
    for i in range(row):
        if board[i] == col or abs(board[i] - col) == abs(i - row):
            return False
    return True

# Solve function
def queen(row):

    if row == N:
        print("Solution:")

        for i in range(N):
            for j in range(N):
                if board[i] == j:
                    print("Q", end=" ")
                else:
                    print(".", end=" ")
            print()
        return

    for col in range(N):
        if check(row, col):
            board[row] = col
            queen(row + 1)

# Main Program
queen(0)
